# Основна лінія: Русинська спадщина та виживання
UZH_missions_1 = {
    slot = 1
    generic = no
    ai = yes
    potential = {
        tag = UZH
    }
    has_country_shield = yes
    

    # ---
    # Місія "Збереження Православної Віри"
    # Механіка: Вимагає православну або католицьку державу з ≥80% єдності релігії та православний храм/собор у власній провінції; запускає подію uzh.1 й ставить прапор `uzh_rite_stabilized`.
    # Наратив: Підтримує обіцянку Свидригайла захищати мукачівських єпископів, тримаючи монастирі та братства під охороною паланків.
    # Референси: Мукачівська єпархія XV ст., Свидригайлові привілеї русинам 1430-х років.
    # ===============================
    # Mission 1: Preserve Orthodox Faith
    # ===============================
    uz_preserve_orthodox_faith = {
        icon = mission_religious
        position = 1
        trigger = {
            OR = {
                religion = orthodox
                religion = catholic
            }
            religious_unity = 0.8
            any_owned_province = {
                religion = orthodox
                OR = {
                    has_building = temple
                    has_building = cathedral
                }
            }
        }

        provinces_to_highlight = { owned_by = ROOT }

        # Додай свій trigger за потреби

        effect = {
            # Пояснювальна підказка
            custom_tooltip = UZH_hungarian_protection_charter_tt

            hidden_effect = {
                set_country_flag = uzh_rite_stabilized
                set_country_flag = uzh_charter_signed

                # 1) Якщо існує Угорщина — зробити її кандидатом у «патрони»
                if = {
                    limit = { HUN = { exists = yes } }
                    HUN = { save_event_target_as = uzh_alternate_patron }
                }
                # 2) Інакше якщо ми субʼєкт достатнього монарха-оверлорда
                else_if = {
                    limit = {
                        is_subject = yes
                        overlord = {
                            government = monarchy
                            total_development = 150
                        }
                    }
                    overlord = { save_event_target_as = uzh_alternate_patron }
                }
                # 3) Інакше спробувати сусіда-монарха з взаємними позитивними думками
                else_if = {
                    limit = {
                        any_neighbor_country = {
                            government = monarchy
                            total_development = 150
                            has_opinion = { who = ROOT value = 100 }
                            ROOT = { has_opinion = { who = PREV value = 80 } }
                        }
                    }
                    random_neighbor_country = {
                        limit = {
                            government = monarchy
                            total_development = 150
                            has_opinion = { who = ROOT value = 100 }
                            ROOT = { has_opinion = { who = PREV value = 80 } }
                        }
                        save_event_target_as = uzh_alternate_patron
                    }
                }
                # 4) Інакше взяти серед великих держав-монархій, з якими ми в союзі і вони нас люблять
                else_if = {
                    limit = {
                        any_country = {
                            is_great_power = yes
                            government = monarchy
                            has_opinion = { who = ROOT value = 120 }
                            ROOT = { alliance_with = PREV }
                        }
                    }
                    random_country = {
                        limit = {
                            is_great_power = yes
                            government = monarchy
                            has_opinion = { who = ROOT value = 120 }
                            ROOT = { alliance_with = PREV }
                        }
                        save_event_target_as = uzh_alternate_patron
                    }
                }

                # Якщо патрона обрали не HUN — трохи підкрутити відносини
                if = {
                    limit = { NOT = { HUN = { exists = yes } } has_saved_event_target = uzh_alternate_patron }
                    add_opinion = {
                        who = event_target:uzh_alternate_patron
                        modifier = opinion_improved_relations
                    }
                    event_target:uzh_alternate_patron = {
                        add_opinion = { who = ROOT modifier = opinion_granted_privileges }
                    }
                }

                # Виклик події з опціями (нагороди «A/B» роби тут у варіантах події)
                country_event = { id = uzh.2 }
            }
        }

        ai_weight = { factor = 500 }
    }

                            is_subject = yes

                        limit = {
                            any_neighbor_country = {
                                government = monarchy
                                total_development = 150
                                has_opinion = { who = ROOT value = 100 }

                    limit = { NOT = { HUN = { exists = yes } } has_saved_event_target = uzh_alternate_patron }
                    add_opinion = { who = event_target:uzh_alternate_patron modifier = opinion_improved_relations }
                    event_target:uzh_alternate_patron = {
                        add_opinion = { who = ROOT modifier = opinion_granted_privileges }

            }
        }
