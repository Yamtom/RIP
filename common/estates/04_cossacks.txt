estate_cossacks = {
	icon = 4
	
	# Used for when declaring independence
	independence_government = republic
	starting_reform = cossacks_reform
	independence_tech_group = eastern

	province_independence_weight = {
		factor = 1
		modifier = {
			factor = 2
			has_terrain = steppe
		}
		modifier = {
			factor = 2
			# This is the correct way to check for multiple cultures
			# to avoid the 'slavic is not a target' errors.
			OR = {
				culture = russian
				culture = ruthenian
				culture = byelorussian
				culture = ukrainian
				culture = polish
				culture = czech
				culture = slovak
				culture = serbian
				culture = croatian
				culture = bulgarian
			}
		}
		modifier = {
			factor = 0.75
			NOT = { religion_group = christian }
		}
		modifier = {
			factor = 1.5
			NOT = { development = 10 }
		}
		modifier = {
			factor = 0.75
			development = 15
		}
		modifier = {
			factor = 0.75
			development = 20
		}
	}

	# If true, country will get estate
	trigger = {	
		has_dlc = "The Cossacks"
		NOT = { has_reform = cossacks_reform }
		OR = {
			# Checks for specific primary cultures
			primary_culture = russian
			primary_culture = ruthenian
			primary_culture = byelorussian
			primary_culture = ukrainian
			# Or if the capital is in the right region
			AND = {
				capital_scope = {
					region = pontic_steppe_region
				}
				religion_group = christian
			}
		}
	}
	
	country_modifier_happy = {
		cavalry_power = 0.2
		cav_to_inf_ratio = 0.2
	}
	country_modifier_neutral = {
		cavalry_power = 0.1
	}
	country_modifier_angry = {
		cavalry_power = -0.1
		land_maintenance_modifier = 0.05
	}
	
	land_ownership_modifier = {
		cossacks_loyalty_modifier = 0.2
	}
	
	color = { 100 50 150 }
}
