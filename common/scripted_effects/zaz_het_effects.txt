# Effects for Zaporizhia and Hetmanate

# Conduct successful raid
zaz_successful_raid_effect = {
	add_treasury = 100
	add_prestige = 5
	add_yearly_manpower = 0.5
	add_country_modifier = {
		name = zaz_successful_raid
		duration = 730
	}
	capital_scope = {
		add_base_manpower = 1
	}
}

# Conduct failed raid
zaz_failed_raid_effect = {
	add_prestige = -5
	add_war_exhaustion = 1
	add_yearly_manpower = -0.3
	add_country_modifier = {
		name = zaz_failed_raid
		duration = 730
	}
}

# Relocate the Sich
zaz_relocate_sich_effect = {
	capital_scope = {
		remove_province_modifier = zaporozhian_sich_capital
	}
	ROOT = {
		capital = 284  # Default Zaporizhia
	}
	capital_scope = {
		add_province_modifier = {
			name = zaporozhian_sich_capital
			duration = -1
		}
		add_base_manpower = 2
		add_base_production = 1
		if = {
			limit = { NOT = { fort_level = 1 } }
			add_building = fort_15th
		}
	}
	add_prestige = 10
	add_stability_or_adm_power = yes
}

# Establish Cossack settlement
establish_cossack_settlement_effect = {
	add_base_tax = 1
	add_base_production = 1
	add_base_manpower = 2
	add_province_modifier = {
		name = cossack_settlement
		duration = -1
	}
	change_culture = ruthenian
	if = {
		limit = { NOT = { religion = orthodox } }
		change_religion = orthodox
	}
}

# Conduct Chaiky raid

	}

		limit = {

			}
		}

			}
		}
	}
}

# Sign Pereiaslav Treaty
het_pereiaslav_treaty_effect = {
	if = {
		limit = {
			exists = RUS
		}
		RUS = {
			create_subject = {
				subject_type = vassal
				subject = ROOT
			}
			add_opinion = {
				who = ROOT
				modifier = het_pereiaslav_oath
			}
		}
		add_country_modifier = {
			name = "het_russian_protection"
			duration = -1
		}
	}
	else_if = {
		limit = {
			exists = MOS
		}
		MOS = {
			create_subject = {
				subject_type = vassal
				subject = ROOT
			}
			add_opinion = {
				who = ROOT
				modifier = het_pereiaslav_oath
			}
		}
		add_country_modifier = {
			name = "het_russian_protection"
			duration = -1
		}
	}
}

# Break with overlord (Mazepa style)
het_break_with_overlord_effect = {
	if = {
		limit = { is_subject = yes }
		overlord = {
			add_opinion = {
				who = ROOT
				modifier = het_mazepa_treason
			}
			add_casus_belli = {
				target = ROOT
				type = cb_vassalize_mission
				months = 120
			}
		}
	}
	add_liberty_desire = 100
	add_prestige = 15
	add_legitimacy = 10
	add_country_modifier = {
		name = "het_independence_struggle"
		duration = 3650
	}
}

# Establish Black Sea trade

	}

			name = cossack_trade_port

		}

	}
}

# Alliance with steppe nomads
alliance_with_nomads_effect = {
	add_country_modifier = {
		name = steppe_alliance
		duration = 3650
	}
	if = {
		limit = {
			any_known_country = {
				OR = {
					culture_group = altaic
					technology_group = nomad_group
				}
				NOT = { war_with = ROOT }
			}
		}
		random_known_country = {
			limit = {
				OR = {
					culture_group = altaic
					technology_group = nomad_group
				}
				NOT = { war_with = ROOT }
			}
			add_opinion = {
				who = ROOT
				modifier = steppe_brotherhood
			}
		}
	}
}
