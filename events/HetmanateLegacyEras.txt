	is_triggered_only = yes

	option = {

		set_country_flag = het_legacy_commonwealth
		add_country_modifier = {
			name = het_legacy_chronicle_commonwealth
			duration = -1
		}
		add_prestige = 10
		ai_chance = {
			factor = 40
		}
	}
	option = {
		name = het_legacy_eras.5.b
		set_country_flag = het_legacy_regimental
		add_country_modifier = {
			name = het_legacy_chronicle_regimental_state
			duration = -1
		}
		add_army_tradition = 10
		ai_chance = {
			factor = 30
		}
	}
	option = {
		name = het_legacy_eras.5.c
		set_country_flag = het_legacy_cultural
		add_country_modifier = {
			name = het_legacy_chronicle_cultural_patronage
			duration = -1
		}
		add_adm_power = 50
		ai_chance = {
			factor = 30
		}
	}

	after = {
		if = {
			limit = {
				NOT = { has_country_flag = het_legacy_pereiaslav_reviewed }
				OR = {
					has_country_flag = pereiaslav_treaty_signed
					has_country_modifier = khmelnitsky_uprising
					has_country_flag = het_legacy_path_khmelnytsky
				}
			}
			country_event = { id = het_legacy_eras.6 days = 20 }
		}
		else_if = {
			limit = {
				NOT = { has_country_flag = het_legacy_mazepa_reviewed }
				OR = {
					has_country_flag = mazepa_independence_attempted
					has_country_modifier = mazepa_golden_age
				}
			}
			country_event = { id = het_legacy_eras.7 days = 20 }
		}
		else_if = {
			limit = {
				NOT = { has_country_flag = het_legacy_collegium_reviewed }
				OR = {
					has_country_flag = collegium_established
					has_country_modifier = het_little_russian_collegium
				}
			}
			country_event = { id = het_legacy_eras.8 days = 20 }
		}
		else = {
			set_country_flag = het_legacy_eras_completed
		}
	}
}
