# =========================================================================
# ZAPOROZHIAN SICH CRISIS DISASTER
# (XVI–XVIII ст.: 1550–1775 р.р. — внутрішні конфлікти й зовнішній тиск)
# =========================================================================
#
# HISTORICAL BACKGROUND:
# The Zaporozhian Sich (Запорізька Січ) was a semi-autonomous, semi-nomadic military community
# organized around democratic principles (rada assembly, elected kosh otaman) and frequent raids.
# Its crises arose from four overlapping sources: (1) internal power struggles (democracy vs. 
# military autocracy), (2) failed or costly raids provoking retaliation, (3) overlord pressure
# (Poland, Moscow, Ottoman) to subordinate the Sich, and (4) Crimean/Ottoman military pressure.
#
# Key Crises & Figures:
# - 1550s–1600s Consolidation: Early Sich emerges as loose confederation of bands; internal
#   discipline weak; frequent disputes over leadership and raid targets; some leaders executed
#   for disobedience by the rada, others dominate the council through military prowess.
# - 1590s–1600s Chaiky Raids: Zaporozhian fleets ("chaiky") raid Black Sea coast (Ottoman and
#   Crimean targets) and capture Kaffa occasionally; Ottoman/Crimean retaliation devastates
#   Sich supply lines; some raids successful, others catastrophic (e.g., 1615 Ottoman expedition
#   against Sich fortifications).
# - 1648–1657 Khmelnytsky Period: Sich supplies elite cossack cavalry and elite infantry to 
#   Khmelnytsky's insurgency; tensions arise between Sich's democratic autonomy and Khmelnytsky's
#   centralizing authority; some Sich leaders resent subordination to hetman's hierarchy.
# - 1657–1709 Ruina Period: Sich splinters into competing factions backing rival hetmans 
#   (Right/Left Bank); some Sich bands raid independently, others are pressed into hetman service;
#   devastating Ottoman/Crimean raids after 1656 hamper Sich recovery; Mazepa attempts to
#   centralize Sich into registered regiments (conflicts with Sich's democratic traditions).
# - 1709 Polтava Aftermath: After Mazepa's failed 1708 alliance with Sweden, Petro I (Peter I)
#   severely restricts Sich autonomy; Russian garrisons positioned nearby; Sich loses independent
#   raiding capacity; increasingly treated as vassal/colonial force under Russian command.
# - 1709–1775 Russian Squeeze: Peter I and successors progressively limit Sich's autonomy;
#   Sich leadership becomes more directly appointed by Russian governors; regulations replace
#   traditions; popular discontent simmers as 1760s–1770s approach.
# - 1775 Final Liquidation: Empress Catherine II orders Sich's dissolution after Ukrainian-Russian
#   War (1768–1774); last remnant (Sich fortifications on island of Khortytsia) stormed by Russian
#   troops; survivors flee to Ottoman lands (Oleshky/Ochakiv region) where they form a new
#   Ottoman-backed "Sich" (later becomes Krasnodar Cossacks on Kuban under Russian rule post-1792).
#
# Internal Democratic vs. Autocratic Tensions:
# - Rada Assembly: Democratic council of all cossacks elected military leaders (kosh otaman, 
#   judge, chancellor); decisions made by consensus or simple majority; elected leaders answerable
#   to body; rotating leadership prevented personal monarchies.
# - Kosh Otaman (Military Commander): During raids or wars, kosh otaman held near-absolute
#   authority for military operations; after campaign, returned to equal status with peers;
#   some ambitious kosh otamans (e.g., Sirko, Doroshenko) attempted to make position hereditary
#   or permanent, provoking internal resistance from egalitarian-minded cossacks.
# - Register vs. Free Cossacks: Polish/Hetman authorities tried to formalize "registered" 
#   cossacks (official, paid, bound by rules) vs. "free" (unregistered, self-supporting, autonomous);
#   Sich's identity was deeply tied to freedom-from-rules; forced registration was seen as 
#   enslavement and provoked revolts within the Sich.
# - Wealth Disparity: Successful raid leaders accumulated private treasuries (loot, slaves,
#   ships); ordinary cossacks often received minimal shares; inequality bred discontent and
#   occasional purges (e.g., execution of corrupt leaders by demokratiya—popular assembly).
#
# External Pressures & Raid Consequences:
# - Polish Crown Demands: Poland's attempts to control and tax the Sich (register it, garrison it)
#   provoked Sich defiance and raids against Polish holdings; successful raids enriched Sich but
#   invited Polish/Lithuanian retaliation; failed raids resulted in casualties and lost ships.
# - Ottoman & Crimean Pressure: Turkish Sultan and Crimean Khan both sought to eliminate or
#   subordinate the Sich; major military expeditions (1588–1595, 1620s, 1656–1657 after Khotyn)
#   inflicted heavy losses; but Sich's raiding also threatened Ottoman tributaries and supply
#   lines, creating mutual strategic pressure.
# - Moscow's Control: After 1654 Pereiaslav, Moscow increasingly viewed Sich as a tool for its
#   southern wars; cossacks pressed into service in Russian-Turkish conflicts; autonomy curtailed
#   through garrisons, regulations, and co-opted leadership.
# - Raid Consequences: Successful raids generated loot and prestige; failures lost ships, men,
#   and supplies, provoking internal blame-shifting and factional accusations of cowardice or
#   corruption. Major failures (e.g., 1625 Ottoman expedition, 1650 Crimean khans) sparked
#   internal purges and crises of legitimacy.
#
# Crisis Resolution Paths (Historical):
# - Authoritarian Consolidation (Mazepa): Attempted to formalize Sich into registered regiments
#   under hetman command; suppressed democratic traditions with force; increased Sich's military
#   power but reduced autonomy; ultimately failed (Poltava, 1709) and ushered in Russian control.
# - Democratic Revival (post-1709): Brief periods where Sich attempted to restore full autonomy
#   from hetman/Russia; elected autonomous kosh otamans, redistributed wealth, reinforced rada
#   authority; suppressed by Russian pressure after 1710s.
# - Foreign Subordination (Ottoman Path, 1709+): Some Sich survivors fled to Ottoman lands after
#   Poltava, founded "Oleshky Sich" under Ottoman protection; became Ottoman frontier force until
#   Russian reconquest of Crimea/Bessarabia (1768–1774) reduced Ottoman Sich to rump state.
# - Annihilation (1775): Catherine II's final solution: military assault on Sich, execution of
#   leadership, prohibition of Sich assembly, redistribution of lands to Russian nobility,
#   incorporation of survivors into Kuban/Black Sea Cossack hosts under tight Russian control.
#
# GAME MECHANICS (Design Notes):
# This disaster models the Sich's perpetual "trapped state": internally democratic but externally
# pressured, free by tradition but militarily subordinate to overlords, autonomous in theory but
# increasingly controlled in practice. It triggers on legitimacy/republican tradition drops (internal
# crisis), war exhaustion (costly raids backfiring), subject status + high liberty desire (foreign
# pressure), or neighbor dominance (Crimean/Ottoman threat). It progresses on weakness and regresses
# on legitimacy/tradition recovery and prestige. The modifier includes increased autonomy 
# (reflecting internal democratic chaos) and increased loot (reflecting ongoing raiding economy),
# but reduced shock damage (reflecting military demoralization). Ending requires restored legitimacy
# or republican tradition, peace, and either independence or low liberty desire.
#
# Mechanics: Disaster for Zaporizhia (ZAZ) representing internal conflicts between
# democratic Sich rada (council) and authoritarian kosh otamans (military leaders),
# raids going wrong, conflicts with overlords (Poland/Russia), and Crimean pressure.
# Causes massive unrest, military power drain, and diplomatic isolation until resolved
# through reform or military victory.
#
# Narrative: Simulates various crises of the Zaporozhian Host including the conflicts
# over autonomy, raids that provoked retaliation, internal power struggles, and the
# final liquidation attempts by Russian Empire (1709, 1775).
#
# References: 
# - Linda Gordon, *Cossack Rebellions* (1983)
# - Victor Ostapchuk, "The Human Landscape of the Ottoman Black Sea" (2001)
# - Evariste Lévi-Provençal, *The Zaporozhian Cossacks* (1967)
# - Serhii Plokhy, *The Cossacks and Religion in Early Modern Ukraine* (2001)
# - Mykhailo Hrushevsky, *History of Ukraine-Rus'*, vols. 7-9
# - Jaroslaw Pelenski, *The Contest for the Legacy of Kievan Rus'* (1998)
# =========================================================================

zaporozhian_sich_crisis = {

	potential = {
		tag = ZAZ
		normal_or_historical_nations = yes
		NOT = { has_country_flag = sich_liquidated }
	}

	can_start = {
		has_any_disaster = no
		OR = {
			current_age = age_of_reformation
			current_age = age_of_absolutism
			current_age = age_of_revolutions
		}
		OR = {
			# Internal crisis
			AND = {
				NOT = { legitimacy = 50 }
				NOT = { republican_tradition = 40 }
				average_autonomy = 30
			}
			# Failed raid consequences
			AND = {
				is_at_war = yes
				war_exhaustion = 5
				NOT = { manpower_percentage = 0.4 }
			}
			# Overlord pressure
			AND = {
				is_subject = yes
				liberty_desire = 60
			}
			# Crimean/Ottoman threat
			AND = {
				any_neighbor_country = {
					OR = {
						tag = CRI
						tag = TUR
					}
					army_size = ROOT
					is_rival = ROOT
				}
			}
		}
	}
	on_start = zaz_crisis.1

	can_stop = {
		has_any_disaster = yes
	}

	modifier = {
		global_unrest = 4
		global_autonomy = 0.05
		land_morale = -0.15
		diplomatic_reputation = -1
		shock_damage = -0.1
		loot_amount = 0.15
	}

	progress = {
		modifier = {
			factor = 2
			always = yes
		}
		modifier = {
			factor = 2
			NOT = { legitimacy = 40 }
		}
		modifier = {
			factor = 2
			NOT = { republican_tradition = 30 }
		}
		modifier = {
			factor = 1.5
			war_exhaustion = 6
		}
		modifier = {
			factor = 1.5
			is_subject = yes
			liberty_desire = 70
		}
		modifier = {
			factor = 1.3
			num_of_rebel_armies = 1
		}
		modifier = {
			factor = 0.75
			stability = 1
		}
		modifier = {
			factor = 0.6
			has_idea_group = defensive_ideas
		}
		modifier = {
			factor = 0.8
			prestige = 50
		}
	}

	can_end = {
		stability = 0
		NOT = { war_exhaustion = 3 }
		OR = {
			AND = {
				legitimacy = 60
			}
			AND = {
				republican_tradition = 50
			}
		}
		NOT = { num_of_rebel_armies = 1 }
		OR = {
			is_subject = no
			NOT = { liberty_desire = 50 }
		}
	}
	on_end = zaz_crisis.2

	on_monthly = {
		random_events = {
			100 = zaz_crisis.3
			80 = zaz_crisis.4
			60 = zaz_crisis.5
		}
	}
}
