# === ІСТОРИЧНИЙ КОНТЕКСТ: XV–XIX ст. Запорозька Січ ===
# Запорізька Січ виникла як напівнезалежна козацька демократична комуна на Нижньому Дніпрі
# (XV ст.), досягла розквіту у XVI–XVII ст. як сила, що впливала на регіональну політику
# (чайки, реєстр, виборність отамана). 1648–1657 — комітет під Б. Хмельницького та розпочинання
# Визвольної вiйни. 1657–1709 — період Руїни, росіїївського закабалення й спроб централізації
# (Мазепа). 1709–1775 — російське закріпачення й остаточна ліквідація Січі 1775 р.
# Вижившi козаки перейшли до Кубанського/Чорноморського козацтва.
#
# МЕХАНІКИ (ідеї для геймплею):
# - Релокація Січі: Томаківська → Базавлуцька → Микитинська → Чортомлицька → Олешківська 
#   (під Османами) → Нова Січ (у 1810–х). Подієва система вибору нової локації зі слоба-бонусами.
# - Чайки/морські рейди: навиковий механіко захоплення й гра ризику проти Османської флоти.
# - Козацький реєстр: механіка формалізації/дикості козаків, відповідь на тиск суверенів.
# - Демократія Січі: виборність отамана (подієві цикли), рада, рівність в дистрибуції здобичі.
# - Старшинські роди: впливові генерали/дипломати як радники й конкуренти.
#
# ІСТОРИЧНІ РЕФЕРЕНСИ:
# - Сергій Плохій, "Брама Європи" (Harvard, 2015); "Cossacks and Religion in Early Modern Ukraine".
# - Олена Апанович, "Збройні сили України першої половини XVIII ст." (1995).
# - Михайло Грушевський, "Історія України-Руси" (т. VII–IX, 1908–1935).
# - Orest Subtelny, "Ukraine: A History" (Toronto, 1988).
# - Frank Sysyn, "Between Poland and the Ukraine" (1985).
#

# The Founding of the Sich
country_event = {
	id = zaz_flavor.2
	title = "zaz_flavor.2.t"
	desc = "zaz_flavor.2.d"
	picture = FORT_eventPicture
	fire_only_once = yes
	
	trigger = {
		tag = ZAZ
		is_year = 1552
		NOT = { is_year = 1560 }
		owns = 284  # Zaporizhia
	}
	
	mean_time_to_happen = {
		months = 24
	}
	
	# Establish Sich on Khortytsia (XV–XVI ст.)
	# Історичний факт: Хортиця як основна база козацьких банд ~ 1550–1560-х;
	# укріплення стану через укріплення острова, тренування, мобілізацію сильних воїнів.
	option = {
		name = "zaz_flavor.2.a"
		284 = {
			add_province_modifier = {
				name = "zaporozhian_sich_founded"
				duration = -1
			}
			add_base_manpower = 2
			add_building = fort_15th
		}
		add_prestige = 15
		add_army_tradition = 10
		add_republican_tradition = 5
		capital_scope = {
			add_province_modifier = {
				name = mobile_sich
				duration = 7300
			}
		}
	}
	
	# Establish mobile Sich (nomadic strategy)
	# Альтернативний історичний шлях: мобільний табір замість фіксованої фортеці;
	# зберігає гнучкість проти противника, але зменшує контроль над територією.
	option = {
		name = "zaz_flavor.2.b"
		capital_scope = {
			add_province_modifier = {
				name = mobile_sich
				duration = 5475
			}
			add_base_manpower = 1
		}
		add_republican_tradition = 10
		add_prestige = 5
	}
}

# The Registered Cossacks Crisis (XVI–XVII ст.)
# Історія: Річ Посполита намагалася формалізувати козацтво через "реєстр" — офіційні,
# платні, дисциплінаційовані сили; "вільні" козаки відстоювали демократичність Січі.
# 1571 — перший реєстр (200 осіб); 1620-і — розширення; 1648 — восстання Хмельницького
# як відповідь на обмеження реєстру й утиски польської шляхти.
country_event = {
	id = zaz_flavor.14
	title = "zaz_flavor.14.t"
	desc = "zaz_flavor.14.d"
	picture = ELECTION_REPUBLICAN_eventPicture
	
	trigger = {
		tag = ZAZ
		is_year = 1620
		NOT = { is_year = 1660 }
		any_neighbor_country = {
			tag = POL
		}
	}
	
	mean_time_to_happen = {
		months = 180
	}
	
	# Expand registered cossacks (more discipline, less freedom)
	# Стратегія: прийняти польські умови й формалізувати армію; бонуси армії, штраф свободи.
	option = {
		name = "zaz_flavor.14.a"
		add_mil_power = 30
		add_republican_tradition = -5
		add_army_tradition = 8
		add_country_modifier = {
			name = "zaz_registered_cossacks"
			duration = 5475
		}
		add_stability_or_adm_power = yes
		POL = {
			add_opinion = {
				who = ROOT
				modifier = opinion_friendly
			}
		}
	}
	
	# Reject registration (maintain independence, risk)
	# Стратегія: охороняти демократичність Січі; рис + ліберті дезайр, конфліктність з Польщею.
	option = {
		name = "zaz_flavor.14.b"
		add_prestige = 10
		add_republican_tradition = 15
		add_liberty_desire = 10
		POL = {
			add_opinion = {
				who = ROOT
				modifier = opinion_hates_us
