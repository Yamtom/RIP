namespace = zaz_crisis

# Sich Crisis Begins
country_event = {
	id = zaz_crisis.1
	title = "zaz_crisis.1.t"
	desc = "zaz_crisis.1.d"
	picture = CIVIL_WAR_eventPicture
	is_triggered_only = yes
	
	immediate = {
		set_country_flag = sich_crisis_active
	}
	
	# Strengthen the Kosh Otaman
	option = {
		name = "zaz_crisis.1.a"
		add_legitimacy = 10
		add_republican_tradition = -5
		add_country_modifier = {
			name = zaz_strong_otaman
			duration = 3650
		}
	}
	
	# Empower the Rada (council)
	option = {
		name = "zaz_crisis.1.b"
		add_republican_tradition = 10
		add_stability = -1
		add_country_modifier = {
			name = zaz_democratic_rada
			duration = 3650
		}
	}
	
	# Launch a great raid to unite the Host
	option = {
		name = "zaz_crisis.1.c"
		add_prestige = 10
		add_yearly_manpower = 0.5
		add_country_modifier = {
			name = zaz_great_raid
			duration = 1825
		}
	}
}

			add_base_manpower = 1
		}
	}
}

# Raid Gone Wrong
country_event = {
	id = zaz_crisis.3
	title = "zaz_crisis.3.t"
	desc = "zaz_crisis.3.d"
	picture = BORDER_TENSION_eventPicture
	is_triggered_only = yes
	
	trigger = {
		has_disaster = zaporozhian_sich_crisis
		any_neighbor_country = {
			OR = {
				tag = CRI
				tag = TUR
				religion_group = muslim
			}
		}
	}
	
	# Accept losses
	option = {
		name = "zaz_crisis.3.a"
		add_prestige = -5
		add_yearly_manpower = -0.3
		add_war_exhaustion = 0.5
	}
	
	# Prepare for retaliation
	option = {
		name = "zaz_crisis.3.b"
		add_mil_power = -25
		random_neighbor_country = {
			limit = {
				OR = {
					tag = CRI
					tag = TUR
				}
			}
			add_opinion = {
				who = ROOT
				modifier = zaz_raid_provocation
			}
		}
	}
}

	title = "zaz_crisis.5.t"

	}
}
