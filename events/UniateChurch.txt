namespace = uniate_church

# =========================================================================
# UNIATE CHURCH EVENT CHAIN (Union of Brest 1596)
# =========================================================================
# These events cover the establishment, development, and consequences
# of the Union of Brest (1596) and subsequent religious dynamics in
# Ruthenian lands under Polish-Lithuanian Commonwealth.
#
# Historical flow:
# 1. Proposal of religious union (Orthodox bishops approach Rome)
# 2. Union of Brest signing (1596)
# 3. Orthodox resistance (brotherhoods, Cossacks oppose)
# 4. Catholic nobility support
# 5. Consolidation or collapse of union
# 6. Long-term consequences (Synod of Zamość 1720, etc.)
# =========================================================================

# Event 1: Proposal for Church Union
country_event = {
	id = uniate_church.1
	title = uniate_church.1.t
	desc = uniate_church.1.d
	picture = RELIGION_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		OR = {
			tag = RPS
			tag = LIT
			tag = KIE
			tag = HLC
			tag = VLN
			tag = PDL
			tag = PLT
			tag = POL
		}
		religion = orthodox
		has_idea_group = religious_ideas
		OR = {
			alliance_with = PAP
			any_neighbor_country = {
				religion = catholic
			}
		}
		NOT = { has_country_flag = uniate_union_proposed }
		year = 1590
		NOT = { year = 1650 }
	}
	
	mean_time_to_happen = {
		months = 48
		modifier = {
			factor = 0.5
			alliance_with = PAP
		}
		modifier = {
			factor = 0.8
			OR = {
				overlord = { religion = catholic }
				senior_union_with = { religion = catholic }
			}
		}
	}
	
	option = { # Accept union proposal
		name = uniate_church.1.a
		ai_chance = { factor = 40 }
		
		set_country_flag = uniate_union_proposed
		set_country_flag = pursuing_uniate_union
		
		add_papal_influence = 10
		add_country_modifier = {
			name = uniate_union_established
			duration = 7300
		}
		
		every_owned_province = {
			limit = { religion = orthodox }
			add_province_modifier = {
				name = provincial_uniate_resistance
				duration = 3650
			}
		}
		
		country_event = { id = uniate_church.2 days = 365 }
	}
	
	option = { # Reject union, preserve Orthodox identity
		name = uniate_church.1.b
		ai_chance = { factor = 60 }
		
		set_country_flag = uniate_union_rejected
		
		add_prestige = 10
		add_country_modifier = {
			name = orthodox_revival_movement
			duration = 5475
		}
	}
}
# Event 2: Union of Brest Signed (1596)
country_event = {
	id = uniate_church.2
	title = uniate_church.2.t
	desc = uniate_church.2.d
	picture = RELIGION_eventPicture
	
	is_triggered_only = yes
	
	option = { # Enforce union aggressively
		name = uniate_church.2.a
		ai_chance = { factor = 30 }
		
		add_country_modifier = {
			name = forced_religious_union
			duration = 5475
		}
		
		every_owned_province = {
			limit = {
				religion = orthodox
				development = 10
			}
			spawn_rebels = {
				type = orthodox_rebels
				size = 1
			}
		}
	}
	
	option = { # Allow gradual transition
		name = uniate_church.2.b
		ai_chance = { factor = 50 }
		
		add_country_modifier = {
			name = dual_communion_compromise
			duration = 7300
		}
		
		random_owned_province = {
			limit = { religion = orthodox }
			change_religion = orthodox # Simulate gradual uniate adoption
			add_base_tax = 1
		}
	}
	
	option = { # Seek Catholic support
		name = uniate_church.2.c
		ai_chance = { factor = 20 }
		
		add_country_modifier = {
			name = catholic_backing_uniate
			duration = 5475
		}
		
		add_papal_influence = 20
		add_curia_treasury = 50
	}

# Event 4: Cossack Orthodox Identity
country_event = {
	id = uniate_church.4
	title = uniate_church.4.t
	desc = uniate_church.4.d
	picture = BATTLE_eventPicture
	
	trigger = {
		has_country_flag = pursuing_uniate_union
		NOT = { has_country_flag = cossack_orthodox_stance_determined }
		OR = {
			has_estate = estate_cossacks
			num_of_owned_provinces_with = {
				value = 3
				culture = ruthenian
				religion = orthodox
			}
		}
	}
	
	mean_time_to_happen = {
		months = 36
	}
	
	option = { # Respect Cossack Orthodox identity
		name = uniate_church.4.a
		ai_chance = { factor = 50 }
		
		set_country_flag = cossack_orthodox_stance_respected
		
		add_country_modifier = {
			name = cossack_orthodox_identity
			duration = 7300
		}
		
		if = {
			limit = { has_estate = estate_cossacks }
			add_estate_loyalty = {
				estate = estate_cossacks
				loyalty = 10
			}
		}
	}
	
	option = { # Force conversion
		name = uniate_church.4.b
		ai_chance = { factor = 30 }
		
		set_country_flag = cossack_orthodox_stance_forced
		
		add_country_modifier = {
			name = forced_religious_union
			duration = 3650
		}
		
		if = {
			limit = { has_estate = estate_cossacks }
			add_estate_loyalty = {
				estate = estate_cossacks
				loyalty = -20
			}
		}
		
		random_owned_province = {
			limit = {
				culture = ruthenian
				religion = orthodox
			}
			spawn_rebels = {
				type = cossack_rebels
				size = 2
			}
		}
	}
	
	option = { # Seek compromise
		name = uniate_church.4.c
		ai_chance = { factor = 20 }
		
		set_country_flag = cossack_orthodox_stance_compromise
		
		add_country_modifier = {
			name = dual_communion_compromise
			duration = 5475
		}
	}
}
