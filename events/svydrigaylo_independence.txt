# Svydrigaylo Independence Path Events
# События для механіки незалежності Рутенської конфедерації
# ID: svydrigaylo.001 - svydrigaylo.103
#
namespace = svydrigaylo

# Event 1: Overlord Notified of Independence Declaration
country_event = {
	id = svydrigaylo.001
	title = "svydrigaylo.001.t"
	desc = "svydrigaylo.001.d"
	picture = CARDINALS_ARGUING_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			# Overlord evaluates response
			if = {
				limit = {
					war_exhaustion = 5
					NOT = { army_strength = { who = FROM value = 1 } }
				}
				set_country_flag = overlord_accepts_independence
			}
			else = {
				set_country_flag = overlord_denies_independence
			}
		}
	}
	
	option = {
		name = "svydrigaylo.001.a"
		trigger = { has_country_flag = overlord_accepts_independence }
		
		FROM = {
			add_opinion = {
				who = ROOT
				modifier = ruthenian_independence_accepted
				years = 20
			}
		}
		add_prestige = 20
		add_stability = 1
	}
	
	option = {
		name = "svydrigaylo.001.b"
		trigger = { has_country_flag = overlord_denies_independence }
		
		FROM = {
			add_casus_belli = {
				type = cb_vassalize_mission
				target = ROOT
			}
		}
		ROOT = {
			add_casus_belli = {
				type = cb_liberation
				target = ROOT
			}
		}
	}
}
