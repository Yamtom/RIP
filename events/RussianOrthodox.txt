# Event 3: Conquest - Forced Conversion Choice

	id = russian_orthodox.3
	title = russian_orthodox.3.t
	desc = russian_orthodox.3.d
	picture = RELIGION_eventPicture
	
	trigger = {
		owner = {
			religion = russian_orthodox
		}
		NOT = { religion = russian_orthodox }
		NOT = { religion = orthodox }
		NOT = { has_province_modifier = forced_conversions }
		is_year = 1500
	}
	
	mean_time_to_happen = {
		months = 60
		modifier = {
			factor = 0.5
			owner = { has_country_modifier = orthodox_inquisition }
		}
		modifier = {
			factor = 1.5
			development = 20
		}
	}
	
	option = { # Force conversion aggressively
		name = russian_orthodox.3.a
		ai_chance = { factor = 60 }
		
		add_province_modifier = {
			name = forced_conversions
			duration = 3650
		}

			chance = 40
			change_religion = russian_orthodox
		}

	}
}
