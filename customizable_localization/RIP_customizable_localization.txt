defined_text = {
	name = GetUZHIdentityCultureSingular
	text = {
		trigger = {
			OR = {
				has_country_flag = chose_ruthenian_identity
				primary_culture = ruthenian
				primary_culture = ruthenian_new
			}
		}
		localisation_key = UZH_CULTURE_SINGULAR_RUTHENIAN
	}
	text = {
		trigger = {
			OR = {

			}
		}

	}
	text = {
		trigger = {
			OR = {
				has_country_flag = chose_rusyn_identity

			}

		localisation_key = UZH_CULTURE_SINGULAR_RUSYN
	}
	text = {
		trigger = { always = yes }

	}
