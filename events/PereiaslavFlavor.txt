namespace = prl_flavor

# The Pereiaslav Council (1654)
country_event = {
	id = prl_flavor.1
	title = "prl_flavor.1.t"
	desc = "prl_flavor.1.d"
	picture = DIPLOMACY_eventPicture
	fire_only_once = yes

	trigger = {
		tag = PRL
		owns = 2407
		is_year = 1654
		NOT = { is_year = 1656 }
		is_at_war = yes
		OR = {
			war_with = PLC
			war_with = POL
		}
		OR = {
			exists = RUS
			exists = MOS
		}
		NOT = { is_subject = yes }
	}

	mean_time_to_happen = {
		months = 6
	}

	option = {
		name = "prl_flavor.1.a"
		set_country_flag = prl_pereiaslav_oath
		if = {
			limit = { exists = RUS }
			RUS = {
				create_subject = {
					subject_type = vassal
					subject = ROOT
				}
				add_opinion = {
					who = ROOT
					modifier = het_pereiaslav_oath
				}
			}
		}
		else_if = {
			limit = { exists = MOS }
			MOS = {
				create_subject = {
					subject_type = vassal
					subject = ROOT
				}
				add_opinion = {
					who = ROOT
					modifier = het_pereiaslav_oath
				}
			}
		}
		add_country_modifier = {
			name = prl_russian_protection
			duration = -1
		}
		add_prestige = -5
	}

	option = {
		name = "prl_flavor.1.b"
		add_prestige = 15
		add_legitimacy = 10
		add_war_exhaustion = -1
		add_country_modifier = {
			name = prl_defiant_council
			duration = 3650
		}
	}
}
