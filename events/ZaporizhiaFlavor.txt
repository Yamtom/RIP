# Clash with Polish-Lithuanian Commonwealth
country_event = {
	id = zaz_flavor.6
	title = "zaz_flavor.6.t"
	desc = "zaz_flavor.6.d"
	picture = DIPLOMATIC_TENSION_eventPicture
	
	trigger = {
		tag = ZAZ
		is_year = 1600
		NOT = { is_year = 1670 }
		any_neighbor_country = {
			tag = POL
		}
	}
	
	mean_time_to_happen = {
		months = 200
		modifier = {
			factor = 0.6
			POL = {
				NOT = { has_opinion = {
					who = ROOT
					value = 0
				}}
			}
		}
	}
	
	# Seek autonomy within Commonwealth
	option = {
		name = "zaz_flavor.6.a"
		if = {
			limit = {
				is_subject = no
			}
			POL = {
				create_subject = {
					subject_type = vassal
					subject = ROOT
				}
			}
		}
		add_country_modifier = {
			name = "zaz_polish_autonomy"
			duration = 3650
		}
		add_liberty_desire = 20
	}
	
	# Resist Polish control
	option = {
		name = "zaz_flavor.6.b"
		add_prestige = 10
		POL = {
			add_opinion = {
				who = ROOT
				modifier = plc_cossack_rebellion
			}
		}
		add_casus_belli = {
			target = POL
			type = cb_insult
			months = 60
		}
	}
	
	# Remain neutral
	option = {
		name = "zaz_flavor.6.c"
		add_stability_or_adm_power = yes
		add_dip_power = 25
	}
}

# The Sich and Russian Tsardom
country_event = {
	id = zaz_flavor.13
	title = "zaz_flavor.13.t"
	desc = "zaz_flavor.13.d"
	picture = DIPLOMACY_eventPicture
	
	trigger = {
		tag = ZAZ
		is_year = 1700
		NOT = { is_year = 1775 }
		OR = {
			exists = RUS
			exists = MOS
		}
		is_subject = no
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	# Accept Russian protection
	option = {
		name = "zaz_flavor.13.a"
		if = {
			limit = { exists = RUS }
			RUS = {
				create_subject = {
					subject_type = vassal
					subject = ROOT
				}
			}
		}
		else_if = {
			limit = { exists = MOS }
			MOS = {
				create_subject = {
					subject_type = vassal
					subject = ROOT
				}
			}
		}
		add_country_modifier = {
			name = zaz_polish_autonomy
			duration = 10950
		}
	}
	
	# Maintain independence
	option = {
		name = "zaz_flavor.13.b"
		add_prestige = 10
		add_republican_tradition = 10
		if = {
			limit = { exists = RUS }
			RUS = {
				add_opinion = {
					who = ROOT
					modifier = zaz_rebellious_cossacks
				}
			}
		}
	}
}
