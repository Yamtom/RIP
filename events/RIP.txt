		}

# The Restoration of Kiev - alt history
country_event = {
	id = flavor_rus.2004
	title = "flavor_rus.EVTNAME2002"
	desc = "flavor_rus.EVTDESC2002"
	picture = COURT_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		has_dlc = "Third Rome"
		NOT = { is_year = 1700 }
		culture_group = east_slavic
		if = {
			limit = { tag = KIE }
			is_year = 1635
		}
		owns = 280
	}

	mean_time_to_happen = { months = 10 }

	option = {
		name = "flavor_rus.EVTOPTA2002"
		if = {
			limit = { 
				NOT = { primary_culture = ruthenian }
			}
			change_primary_culture = ruthenian
		}
		280 = {
			move_capital_effect = yes
			add_base_tax = 4
			add_base_production = 3
			add_base_manpower = 3
		}
		ai_chance = {
			factor = 12
		}
	}

	option = {
		name = "flavor_rus.EVTOPTB2002"
		trigger = { 280 = { is_capital = no } }
		capital_scope = {
			add_base_tax = 1
			add_base_production = 2
			add_base_manpower = 2
		}
		ai_chance = {
			factor = 88
		}
	}
}
