country_event = {
	id = M_UKR_Events.11
	title = M_UKR_Events.11.t
	desc = M_UKR_Events.11.desc
	
	picture = RELIGION_eventPicture


	option = {
		name = M_UKR_Events.11.a		# A Local Noble
		if = {
			limit = {
				has_estate = estate_nobles
			}
			add_estate_loyalty = {
				estate = estate_nobles
				loyalty = 15
			}
			set_dynasty = event_target:noble_province
		}
		if = {
			limit = {
				has_saved_event_target = noble_province
				NOT = {
					has_global_modifier_value = {
						which = female_advisor_chance
						value = 1
					}
				}
			}
			set_dynasty = {
				age = 35
				hidden = yes
				culture = event_target:noble_province
			}
		}
		add_devotion = -5
		add_prestige = 10
		clr_country_flag = in_theocracy_heir_selection
		custom_tooltip = theocracy.1.tt
	}	
	option = {
		name = M_UKR_Events.11.b		# A Foreign Noble
		trigger = { 
			any_neighbor_country = {
				government = monarchy
				has_regency = no
			}
		}
		if = {
			limit = {
				has_estate = estate_nobles
			}
			add_estate_loyalty = {
				estate = estate_nobles
				loyalty = 10
			}
		}
		random_neighbor_country = { 
			limit = {
				government = monarchy
				has_regency = no
				religion = ROOT
			}
			ROOT = { 
				if = {
					limit = {
						NOT = {
							has_global_modifier_value = {
								which = female_advisor_chance
								value = 1
							}
						}
					}
					set_dynasty = { 
						dynasty = PREV 
						age = 35
						culture = PREV
						hidden = yes
					} 
				}
				else = {
					set_dynasty = { 
						dynasty = PREV 
						age = 35
						culture = PREV
						female = yes
						hidden = yes
					} 
				}
			}
			add_opinion = { who = ROOT modifier = opinion_theocracy_noble }
		}
		clr_country_flag = in_theocracy_heir_selection
		custom_tooltip = theocracy.1.tt
	}
