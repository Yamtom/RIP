# Event chain: Polesian Wetland Defense
polesian_belarus.1 = {
	# Marsh Warfare Tradition
	title = polesian_belarus.1.t
	desc = polesian_belarus.1.d
	picture = LAND_MILITARY_eventPicture
	
	trigger = {
		OR = {
			tag = POL
			tag = TRV
			primary_culture = polesian
		}
		any_owned_province = {
			has_terrain = marsh
		}
		NOT = { has_country_modifier = polesian_marsh_tactics }
	}
	
	mean_time_to_happen = {
		months = 120
	}
	
	option = {
		# Train our soldiers in marsh warfare
		name = polesian_belarus.1.a
		add_country_modifier = {
			name = polesian_marsh_tactics
			duration = 7300
		}
		add_mil_power = -25
	}
	
	option = {
		# Focus on conventional warfare
		name = polesian_belarus.1.b
		add_army_tradition = 5
	}
}

# Belarusian Unification Movement
polesian_belarus.3 = {
	# Proto-nationalist sentiment
	title = polesian_belarus.3.t
	desc = polesian_belarus.3.d
	picture = REFORM_eventPicture
	
	trigger = {
		OR = {
			tag = PLT
			tag = MSK
			tag = TRV
			tag = VIT
			tag = MSL
		}
		primary_culture = byelorussian
		is_subject = no
		NOT = { exists = BLR }
		num_of_owned_provinces_with = {
			value = 5
			culture = byelorussian
		}
	}
	
	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.8
			is_year = 1550
		}
		modifier = {
			factor = 0.7
			is_year = 1600
		}
	}
	
	option = {
		# Support unification
		name = polesian_belarus.3.a
		add_country_modifier = {
			name = belarusian_unity_movement
			duration = 7300
		}
		every_province = {
			limit = {
				culture = byelorussian
				NOT = { owned_by = ROOT }
			}
			add_permanent_claim = ROOT
		}
	}
	
	option = {
		# Maintain current borders
		name = polesian_belarus.3.b
		add_stability_or_adm_power = yes
	}
}
