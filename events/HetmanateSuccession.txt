namespace = het_succession

# Hetman Succession Crisis Begins
country_event = {
	id = het_succession.1
	title = "het_succession.1.t"
	desc = "het_succession.1.d"
	picture = CIVIL_WAR_eventPicture
	is_triggered_only = yes
	
	immediate = {
		set_country_flag = hetman_succession_active
	}
	
	# Support strong hetman
	option = {
		name = "het_succession.1.a"
		add_legitimacy = 15
		add_absolutism = 5
		add_country_modifier = {
			name = het_strong_hetman
			duration = 3650
		}
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -10
		}
	}
	
	# Compromise with starshyna
	option = {
		name = "het_succession.1.b"
		add_stability = -1
		add_country_modifier = {
			name = het_starshyna_council
			duration = 3650
		}
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = 15
		}
	}
	
	# Seek foreign backing
	option = {
		name = "het_succession.1.c"
		trigger = {
			any_neighbor_country = {
				OR = {
					tag = POL
					tag = RUS
					tag = MOS
				}
			}
		}
		add_prestige = -10
		add_liberty_desire = 10
		add_country_modifier = {
			name = het_foreign_backing
			duration = 3650
		}
	}
}

# Succession Crisis Ends
country_event = {
	id = het_succession.2
	title = "het_succession.2.t"
	desc = "het_succession.2.d"
	picture = COURT_eventPicture
	is_triggered_only = yes
	
	immediate = {
		clr_country_flag = hetman_succession_active
		set_country_flag = succession_resolved
	}
	
	# New hetman consolidates power
	option = {
		name = "het_succession.2.a"
		add_legitimacy = 20
		add_prestige = 15
		add_country_modifier = {
			name = het_consolidated_power
			duration = 5475
		}
	}
	
	# Reformed election system
	option = {
		name = "het_succession.2.b"
		change_government_reform_progress = 50
		add_country_modifier = {
			name = "het_reformed_election"
			duration = 5475
		}
	}
	
	# Balance between hetman and starshyna
	option = {
		name = "het_succession.2.c"
		add_stability = 1
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = 10
		}
		add_estate_loyalty = {
			estate = estate_burghers
			loyalty = 10
		}
	}
}

# Rival Hetman Emerges
country_event = {
	id = het_succession.3
	title = "het_succession.3.t"
	desc = "het_succession.3.d"
	picture = CIVIL_WAR_eventPicture
	is_triggered_only = yes
	
	trigger = {
		has_disaster = hetman_succession_crisis
	}
	
	# Support current hetman
	option = {
		name = "het_succession.3.a"
		add_mil_power = -50
		add_legitimacy = 10
		random_owned_province = {
			limit = { is_state_core = ROOT }
			spawn_rebels = {
				type = pretender_rebels
				size = 2
				leader = "Rival Hetman"
			}
		}
	}
	
	# Hold new election
	option = {
		name = "het_succession.3.b"
		add_prestige = -10
		define_ruler = {
			change_adm = 1
			change_dip = 1
			change_mil = 3
		}
	}
	
	# Let them duel
	option = {
		name = "het_succession.3.c"
		add_stability = -1
		random = {
			chance = 50
			add_legitimacy = 10
			add_prestige = 5
		}
	}
}

# Foreign Interference
country_event = {
	id = het_succession.5
	title = "het_succession.5.t"
	desc = "het_succession.5.d"
	picture = DIPLOMACY_eventPicture
	is_triggered_only = yes
	
	trigger = {
		has_disaster = hetman_succession_crisis
		is_subject = yes
		overlord = {
			OR = {
				tag = POL
				tag = RUS
				tag = MOS
			}
		}
	}
	
	# Accept overlord's candidate
	option = {
		name = "het_succession.5.a"
		add_liberty_desire = -20
		add_prestige = -10
		define_ruler = {
			change_adm = -1
			change_dip = 2
			change_mil = 1
		}
		overlord = {
			add_opinion = {
				who = ROOT
				modifier = het_obedient_hetmanate
			}
		}
	}
	
	# Resist interference
	option = {
		name = "het_succession.5.b"
		add_liberty_desire = 25
		add_prestige = 10
		add_legitimacy = 10
		overlord = {
			add_opinion = {
				who = ROOT
				modifier = het_defiant_hetmanate
			}
			add_casus_belli = {
				target = ROOT
				type = cb_vassalize_mission
				months = 60
			}
		}
	}
	
	# Negotiate autonomy
	option = {
		name = "het_succession.5.c"
		add_dip_power = -75
		add_legitimacy = 5
		add_liberty_desire = 10
	}
}
