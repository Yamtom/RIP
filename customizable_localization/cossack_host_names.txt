# =========================================================================
# COSSACK STATES "HOST" NAMING SYSTEM
# =========================================================================
# HISTORICAL CONTEXT:
# All major Cossack military formations used "Host" (вiйсько/войско) designation:
# - Zaporozhian Host (Військо Запорозьке)
# - Don Host (Донское войско)  
# - Kuban Host (Кубанское войско)
# - Terek Host (Терское войско)
# - Black Sea Host (Черноморское войско)
#
# This reflects their military-administrative nature as autonomous armed forces
# rather than traditional states. The term emphasized their status as permanent
# military societies organized for frontier defense and raiding.
#
# IMPLEMENTATION:
# Uses EU4's customizable localization system to dynamically add "Host" suffix
# to Cossack state names based on government reforms and culture.
#
# REFERENCES:
# - Brian Boeck, "Imperial Boundaries" (2009), Ch. 3
# - Shane O'Rourke, "Warriors and Peasants" (2000), pp. 15-45
# - Albert Seaton, "The Horsemen of the Steppes" (1985)
# =========================================================================

# Rank-based titles for Cossack Hosts
defined_text = {
	name = GetCossackHostRank
	
	# Rank 1: Local Host
	text = {
		trigger = {
			government_rank = 1
			OR = {
				has_reform = cossacks_reform
				has_reform = zaz_sich_brotherhood_reform
			}
		}
		localisation_key = COSSACK_HOST_RANK_1
	}
	
	# Rank 2: Great Host
	text = {
		trigger = {
			government_rank = 2
			OR = {
				has_reform = cossacks_reform
				has_reform = zaz_sich_brotherhood_reform
			}
		}
		localisation_key = COSSACK_HOST_RANK_2
	}
	
	# Rank 3: Imperial Host
	text = {
		trigger = {
			government_rank = 3
			OR = {
				has_reform = cossacks_reform
				has_reform = zaz_sich_brotherhood_reform
			}
		}
		localisation_key = COSSACK_HOST_RANK_3
	}
	
	# Default
	text = {
		localisation_key = ""
	}
}
