country_event = {
	id = polesian_belarus.20
	title = polesian_belarus.20.t
	desc = polesian_belarus.20.d
	picture = CULTURE_eventPicture
	is_triggered_only = yes
	
	trigger = {
		tag = BLR
	}
	
	option = {
		name = polesian_belarus.20.a
		trigger = {
			OR = {
				primary_culture = byelorussian
				primary_culture = ruthenian
			}
		}
		ai_chance = { factor = 60 }
		
		if = {
			limit = {
				NOT = { primary_culture = byelorussian }
			}
			change_primary_culture = byelorussian
		}
		
			every_owned_province = {
				limit = {
					culture = ruthenian
				}
				change_culture = byelorussian
			}		add_prestige = 15
		add_stability_or_adm_power = yes
		add_country_modifier = {
			name = belarus_national_identity
			duration = 7300
		}
	}
	
	option = {
		name = polesian_belarus.20.b
		trigger = {
			OR = {
				primary_culture = byelorussian
				primary_culture = ruthenian
			}
		}
		ai_chance = { factor = 30 }
		
		if = {
			limit = {
				NOT = { primary_culture = ruthenian }
			}
			change_primary_culture = ruthenian
		}
		
		every_owned_province = {
			limit = {
				culture = byelorussian
			}
			change_culture = ruthenian
		}
		
		add_prestige = 10
		add_republican_tradition = 5
		add_country_modifier = {
			name = belarus_ruthenian_heritage
			duration = 7300
		}
	}
	
	option = {
		name = polesian_belarus.20.c
		ai_chance = { factor = 10 }
		
		add_stability_or_adm_power = yes
		add_dip_power = 20
		add_country_modifier = {
			name = belarus_mixed_heritage
			duration = -1
		}
	}
}
