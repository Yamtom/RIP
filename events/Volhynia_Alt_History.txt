# Event 1: Eastern Partnership — Carpathian Union (Austrian analogue)
country_event = {
	id = rip_vol_alt_history.1
	title = rip_vol_alt_history.1.t
	desc = rip_vol_alt_history.1.d
	picture = DIPLOMACY_eventPicture
	is_triggered_only = yes
	
	option = {
		name = rip_vol_alt_history.1.a # Accept Carpathian Federation
		trigger = {
			has_country_flag = rip_vol_eastern_path
			alliance = HLC
		}
		set_country_flag = rip_vol_carpathian_partner
		add_country_modifier = {
			name = vol_carpathian_integration
			duration = -1
		}
		HLC = {
			add_opinion = {
				who = ROOT
				modifier = vol_carpathian_brother
				years = 50
			}
		}
		add_adm_power = 100
	}
	
	option = {
		name = rip_vol_alt_history.1.b # Pursue Independent Path
		trigger = {
			has_country_flag = rip_vol_eastern_path
		}
		clr_country_flag = rip_vol_eastern_path
		set_country_flag = rip_vol_independent
		add_prestige = -10
		add_adm_power = 50
	}
}

# Event 5: VOL-HLC Galician Union Proposal (mirrors PLC or AUH formation)
country_event = {
	id = rip_vol_alt_history.5
	title = rip_vol_alt_history.5.t
	desc = rip_vol_alt_history.5.d
	picture = DIPLOMACY_eventPicture
	is_triggered_only = yes
	fire_only_once = yes
	
	trigger = {
		tag = VOL
		exists = HLC
		HLC = {
			num_of_owned_provinces = 10
		}
		num_of_owned_provinces = 15
		alliance = HLC
		has_country_flag = rip_vol_carpathian_partner
		OR = {
			HLC = { has_country_flag = rip_galicia_path_austria }
			HLC = { has_country_flag = rip_galicia_path_poland }
		}
	}
	
	mean_time_to_happen = {
		months = 24
		modifier = {
			factor = 0.8
			has_country_flag = rip_vol_eastern_path
		}
	}
	
	option = {
		name = rip_vol_alt_history.5.a # Form Carpathian Union
		set_country_flag = rip_vol_carpathian_union_formed
		HLC = {
			set_country_flag = hlc_carpathian_union_formed
			add_opinion = {
				who = ROOT
				modifier = vol_eternal_union
				years = 100
			}
		}
		create_union = HLC
		add_prestige = 100
		add_adm_power = 200
	}
	
	option = {
		name = rip_vol_alt_history.5.b # Maintain Alliance Only
		add_opinion = {
			who = HLC
			modifier = vol_reject_union
			years = 20
		}
		add_prestige = 25
	}
}

# Event 8: Confessional Crisis — Synod of Ruthenia
country_event = {
	id = rip_vol_alt_history.8
	title = rip_vol_alt_history.8.t
	desc = rip_vol_alt_history.8.d
	picture = RELIGION_eventPicture
	is_triggered_only = yes
	
	trigger = {
		tag = VOL
		OR = {
			has_country_flag = rip_vol_confessional_orthodox
			has_country_flag = rip_vol_confessional_uniate
			has_country_flag = rip_vol_confessional_plural
		}
		religious_unity = 0.6
		current_age = age_of_enlightenment
	}
	
	mean_time_to_happen = {
		months = 36
	}
	
	option = {
		name = rip_vol_alt_history.8.a # Orthodox Consolidation
		if = {
			limit = { has_country_flag = rip_vol_confessional_orthodox }
			add_country_modifier = {
				name = vol_orthodox_consolidated
				duration = 7300
			}
			religious_unity = 0.1
		}
		if = {
			limit = { has_country_flag = rip_vol_confessional_uniate }
			add_stability = -1
		}
	}
	
	option = {
		name = rip_vol_alt_history.8.b # Protect Uniate Autonomy
		if = {
			limit = { has_country_flag = rip_vol_confessional_uniate }
			add_country_modifier = {
				name = vol_uniate_protected
				duration = 7300
			}
			add_papal_influence = 100
		}
		if = {
			limit = { has_country_flag = rip_vol_confessional_orthodox }
			add_stability = -1
		}
	}
	
	option = {
		name = rip_vol_alt_history.8.c # Religious Tolerance Edict
		if = {
			limit = { has_country_flag = rip_vol_confessional_plural }
			add_country_modifier = {

			}

		}

	}
