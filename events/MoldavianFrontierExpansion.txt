country_event = {
	id = moldavian_frontier.3
	title = moldavian_frontier.3.t
	desc = moldavian_frontier.3.d
	picture = TRADEGOODS_eventPicture
	is_triggered_only = yes

	option = {
		name = moldavian_frontier.3.a
		set_country_flag = mol_path_trade_customs
		add_country_modifier = {
			name = moldavian_trade_customs
			duration = 7300
		}
		if = {
			limit = { owns = 282 }
			282 = {
				add_province_modifier = {
					name = moldavian_customs_office
					duration = 9125
				}
			}
		}
		if = {
			limit = {
				OR = {
					has_country_flag = dniester_wine_trade_established
					has_country_flag = dniester_salt_established
				}
			}
			add_mercantilism = 1
		}
		else = {
			add_dip_power = 30
		}
		ai_chance = {
			factor = 60
			modifier = {
				factor = 1.4
				trade_income_percentage = 0.25
			}
		}
	}
	option = {
		name = moldavian_frontier.3.b
		set_country_flag = mol_path_host_settlements
		add_country_modifier = {
			name = moldavian_host_settlements
			duration = 7300
		}
		every_owned_province = {
			limit = {
				OR = {
					area = moldavia_area
					AND = {
						region = crimea_region
						has_terrain = steppe
					}
				}
			}
			add_province_modifier = {
				name = moldavian_waystation_node
				duration = 3650
			}
		}
		add_yearly_manpower = 0.5
		if = {
			limit = { has_country_flag = dniester_estuary_developed }
			add_prestige = 5
		}
		ai_chance = {
			factor = 40
			modifier = {
				factor = 1.25
				manpower_percentage = 0.4
			}
		}
	}
}
